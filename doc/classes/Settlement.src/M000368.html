  def link
    if exists?
      desc = description
      if desc.length > 140
        desc = description.slice(0,140) + "..." end
      "<a href='settlement.cgi?id=#{mysql_id}' " +
      "title='#{desc}' " +
      ">#{name}</a>"
    else
      '<i>None</i>'
    end
  end
