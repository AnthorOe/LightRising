  def to_t
    secs = self.to_i.abs
    if secs < 60
      return "#{secs} seconds"
    elsif secs < 3600
      mins = secs / 60
      return "#{mins} minutes"       
    elsif secs < 3600 * 24
      hours = secs / 3600
      return "#{hours} hours" 
    else
      days = secs / (3600 * 24)
      return "#{days} days"
    end
  end
