def html_select_building(user_id = nil)
  html = "\n\t<select name=\"building\" style=\"width:10em\">\n"
  buildings = db_table(:building).values
  buildings = buildings.select {|building| yield(building)} if block_given?
  buildings.each {
    |building|
    html += "\t\t<option value=\"#{building[:id]}\">"
    html += "#{describe_craft(building)}</option>\n"
  }
  html += "\t</select>\n"
end
