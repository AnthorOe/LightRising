def tile_dir(start_x, start_y, start_z, target_x, target_y)
# When I'm standing on start_x, start_y, start_z,

# What direction should appear on tile target_x, target_z?


  x_offset = target_x - start_x
  y_offset = target_y - start_y

  if start_z == 0 
    if x_offset == 0 && y_offset == 0
      if valid_location?(target_x, target_y, 1) then 'Enter'
      else nil
      end
    else
      offset_to_dir(x_offset,y_offset,0)
    end
  else
    if x_offset == 0 && y_offset == 0 then 'Exit'
    else nil
    end
  end
end
