def stockpile_item_amount(x, y, item_id)
  query = "SELECT amount FROM `stockpiles`" +
  mysql_where({'x'=>x, 'y'=>y, 'item_id'=>item_id})
  
  result = $mysql.query(query)
  if result.num_rows != 0
    result.fetch_row[0].to_i # = result['amount']

  else
    0
  end
end
