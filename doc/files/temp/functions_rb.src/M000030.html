def describe_building(x, y, z, building_id=nil)
  building = Building.new(x, y)
  if building_id == nil then building_id = building.building_id end
  if building_id == nil || building_id == 0 then return '' end
  desc =
  if z == 0
    "There is #{a_an(db_field(:building, building_id, :name))} here"
  else
    "You are inside #{a_an(db_field(:building, building_id, :name))}"
  end
  if building_id == 3 # 3 -> stockpile

    desc += ', containing <span class="small">'
    contents, _ = html_inventory(x, y, " ", :commas, :inline)
    desc += contents + '</span>'
  end
  desc += '.'
  writing = building.writing(z)
  if writing
    if z == 0
      desc += " Written on #{building.name} are the words " +
      "<i>\"#{writing}\"</i>"
    else
      desc += " Written on the wall are the words " +
      "<i>\"#{writing}\"</i>"
    end
  end
  desc
end
