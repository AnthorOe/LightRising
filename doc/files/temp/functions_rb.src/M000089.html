def mysql_change_ap(user, change)

  user_id = 
  if user.kind_of?(Integer) || user.kind_of?(String)
    user
  else
    user.mysql_id
  end

  if change > 0
    mysql_bounded_update('users','ap',user_id,change,Max_AP)
  else
    mysql_bounded_update('users','ap',user_id,change,-Max_AP)  
  end
end
