def html_location_box(user)
  tile = Tile.new(user.x, user.y)
  html = '<div class="locationbox">'
  if tile.settlement
    html += tile.settlement.link + ', ' end
  begin
    html += db_field(:region, tile.region_id, :name)
  rescue Exception => e
    html += 'The Wilderness'
  end
  html += ' - ' + 
  season.to_s + 
  '</div>'
  html
end
