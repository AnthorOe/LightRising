def describe_number(n)
  case n.to_i
    when 0 then 'no'
    when 1 then 'one'
    when 2 then 'two'
    when 3 then 'three'
    when 4 then 'four'
    when 5 then 'five'
    when 6 then 'six'
    when 7 then 'seven'
    when 8 then 'eight'
    when 9 then 'nine'
    when 10 then 'ten'
    when 11 then 'eleven'
    when 12 then 'twelve'
    when 13 then 'thirteen'
    when 14 then 'fourteen'
    when 15 then 'fifteen'
    when 16 then 'sixteen'
    when 17 then 'seventeen'
    when 18 then 'eighteen'
    when 19 then 'nineteen'
    when 20 then 'twenty'
    when (21..29) then 'twenty-' + describe_number(n - 20)
    when 30 then 'thirty'
    when (31..39) then 'thirty-' + describe_number(n - 30)
    when 40 then 'forty'
    when (41..49) then 'forty-' + describe_number(n - 40)
    else 'loads of'
  end
end
