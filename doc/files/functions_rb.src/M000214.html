def html_map(x, y, range, z=0, buttons=false)
# generate html to display a map centered on x, y, and all tiles within

# 'range' of centre tile. Eg range = 2, map size = 5x5.

# If buttons is true, map will include move buttons.

# z determines which buttons are shown, according to tile_dir

# (eg, can't move N/NW/etc if inside)


  html = (-range..range).map {
    |y_offset| "\n\t<tr>\n" + (-range..range).map {
      |x_offset|
        "\t\t" + 
        html_tile((x + x_offset), (y + y_offset), z, x_offset, y_offset, buttons)
      }.to_s + "\n\t</tr>\n"
    }.to_s
  "\n<table>" + html + "\n</table>\n"
end
