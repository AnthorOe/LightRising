def tile_image(terrain)
  image = db_field(:terrain, terrain, :image)
  if image.kind_of? Hash
    if image[season] != nil
      image = image[season]
    else
      image = image[:default] end end
  if image == nil
    image = db_field(:terrain, :wilderness, :image) end
  image
end
